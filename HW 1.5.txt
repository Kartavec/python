R = float(input("Ввеите размер выручки за период "))
C = float(input("Ввеите размер расходов за период "))
P = R - C
margin = P / R * 100
if P >= 0:
    print (f"Ваша фирма сгенерировала прибыль в размере {P} руб.")
    print ("Ренетабильность продаж составила %.2f" % margin, "%")
    Q = int(input("Ввеите коичестов сотрудников "))
    PS = P / Q
    print("Прибыль на 1-го сотрудника составила %.2f" % PS, "руб")
else:
    print (f"Ваша фирма сгенерировала убыток в размере {P} руб.")
